## create rule set for tiva launchpad
##SUBSYSTEM=="tty", ATTRS{idVendor}=="1cbe", ATTRS{idProduct}=="00fd", GROUP="dialout", MODE="0666"
sudo -s

sudo systemctl restart udev 
sudo udevadm trigger
sudo udevadm settle
